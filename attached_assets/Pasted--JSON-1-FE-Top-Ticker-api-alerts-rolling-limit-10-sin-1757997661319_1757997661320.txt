스모크 테스트(반드시 실행·로그 남겨)
# 블로그 랭크 체크 (임의 target_ids로)
curl -sX POST http://localhost:3000/api/rank/blog/check -H "Content-Type: application/json" \
 -d '{"target_ids":[],"query_override":["홍삼","홍삼스틱"]}' | jq .

# 히스토리 (exposed, streak_days 포함 여부)
curl -s "http://localhost:3000/api/rank/history?kind=blog&target_id=<ID>&range=14d" | jq '.[0]'

# 키워드 조회량/점수
curl -s "http://localhost:3000/api/keywords/lookup?texts=홍삼,홍삼스틱" | jq .

# 그룹 지수
curl -s "http://localhost:3000/api/groups/<gid>/index?range=30d" | jq .

# DB 자동중지 규칙
curl -s "http://localhost:3000/api/db/collection-rules" | jq .
